#include<bits/stdc++.h>
using namespace std;

int board[11][11];

bool isPossible(int n,int r,int c){
    
    for(int j = r-1; j >= 0; j--)
        if(board[j][c] == 1)
            return false;
 
    for(int i = r-1, j = c-1; i >= 0 && j >= 0; i--, j--)
        if(board[i][j] == 1)
            return false;
  
    for(int i = r-1, j = c+1; i >= 0 && j < n; i--, j++)
        if(board[i][j] == 1)
            return false;
    
    return true;
}

void placeRow(int n,int r){
    
    if(r == n){
        for(int i = 0; i < n; i++)
            for(int j = 0; j < n; j++)
                cout << board[i][j] << " ";
        cout << endl;
    }
    
    for(int j = 0; j < n; j++){
        
        if(isPossible(n,r,j)){
            
            board[r][j] = 1;
            placeRow(n, r+1);
            board[r][j] = 0;   
        }      
    }
}

void placeNQueens(int n){
  memset(board, 0, 11*11*sizeof(int));  
  placeRow(n, 0);
}

int main(){
    
    int n;
    cin >> n;
    placeNQueens(n);
    
}
